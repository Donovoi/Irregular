# A property within a JSON string
(?:         # Match but don't store:
[\{\,]      # A bracket or comma
\s{0,}      # Optional Whitespace
"           # the opening quote
)            
(?<Name>    # Capture the Name, which is:
.+?         # Anything until...
(?=(?<!\\)")# a closing quote (as long as it's not preceeded by a \) 
)
(?:         # Match but don't store:
"\s{0,}     # a double-quote, optional whitespace:
)
:
?<JSON_Value>
